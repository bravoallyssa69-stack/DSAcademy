Pseudocode Library

1. Bubble Sort

REPEAT until no swaps:
  FOR i = 0 to n-1:
    IF A[i] > A[i+1]:
       SWAP A[i], A[i+1]

2. Selection Sort

FOR i = 0 to n-1:
  min = i
  FOR j = i+1 to n:
    IF A[j] < A[min] THEN min = j
  SWAP A[i], A[min]

3. Insertion Sort

FOR i = 1 to n:
  key = A[i]
  j = i - 1
  WHILE j >= 0 AND A[j] > key:
    A[j+1] = A[j]
    j = j - 1
  A[j+1] = key

4. Linear Search

FOR EACH element IN list:
  IF element == target:
     RETURN index
RETURN -1

5. Binary Search

WHILE low <= high:
  mid = (low + high) / 2
  IF arr[mid] == x: RETURN mid
  ELSE IF arr[mid] < x: low = mid + 1
  ELSE: high = mid - 1

6. Factorial

FUNCTION factorial(n):
  IF n == 0 RETURN 1
  RETURN n * factorial(n-1)

7. Fibonacci

FUNCTION fib(n):
  IF n <= 1 RETURN n
  RETURN fib(n-1) + fib(n-2)

8. Quick Sort

FUNCTION quickSort(arr, low, high):
  IF low < high:
    p = partition(arr, low, high)
    quickSort(arr, low, p - 1)
    quickSort(arr, p + 1, high)

9. Merge Sort

FUNCTION mergeSort(arr):
  IF length < 2 RETURN
  mid = length / 2
  L = mergeSort(arr[0..mid])
  R = mergeSort(arr[mid..end])
  merge(L, R)

10. Palindrome Check

rev = reverse(str)
IF str == rev:
  RETURN True
ELSE:
  RETURN False

12. DFS Graph

FUNCTION DFS(node):
  visited[node] = true
  PRINT node
  FOR EACH neighbor OF node:
    IF neighbor NOT visited:
       DFS(neighbor)
