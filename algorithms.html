1. Bubble Sort

O(n²)

Swaps adjacent elements if they are in wrong order.

void bubbleSort(int arr[], int n) {
    for(int i=0; i<n-1; i++)     
        for(int j=0; j<n-i-1; j++) 
            if(arr[j] > arr[j+1]) swap(arr[j], arr[j+1]); 
}

2. Selection Sort

O(n²)

Finds minimum element and puts it at beginning.

void selectionSort(int arr[], int n) {
    for(int i=0; i<n-1; i++) {
        int min_idx = i;
        for(int j=i+1; j<n; j++)
            if(arr[j] < arr[min_idx]) min_idx = j;
        swap(arr[min_idx], arr[i]);
    }
}

3. Insertion Sort

O(n²)

Builds sorted array one item at a time.

void insertionSort(int arr[], int n) {
    for(int i=1; i<n; i++) {
        int key = arr[i], j = i-1;
        while(j>=0 && arr[j]>key) {
            arr[j+1] = arr[j]; j--;
        }
        arr[j+1] = key;
    }
}

4. Linear Search

O(n)

Checks every element one by one.

int search(int arr[], int n, int x) {
    for(int i=0; i<n; i++)
        if(arr[i]==x) return i;
    return -1;
}

5. Binary Search

O(log n)

Efficient search on sorted arrays.

int binarySearch(int arr[], int l, int r, int x) {
    while(l <= r) {
        int m = l + (r - l) / 2;
        if(arr[m] == x) return m;
        if(arr[m] < x) l = m + 1;
        else r = m - 1;
    }
    return -1;
}

6. Factorial

O(n)

Calculates n! using recursion.

int factorial(int n) {
    if(n == 0) return 1;
    return n * factorial(n - 1);
}

7. Fibonacci

O(2ⁿ)

Generates the nth Fibonacci number.

int fib(int n) {
    if(n <= 1) return n;
    return fib(n-1) + fib(n-2);
}

8. Quick Sort

O(n log n)

Divide and Conquer with pivot.

int partition(int arr[], int l, int h) {
    int pivot = arr[h], i = (l - 1);
    for(int j=l; j<h; j++) {
        if(arr[j] < pivot) {
            i++; swap(arr[i], arr[j]);
        }
    }
    swap(arr[i + 1], arr[h]);
    return (i + 1);
}

9. Merge Sort

O(n log n)

Recursively merges sorted subarrays.

void mergeSort(int arr[], int l, int r) {
    if(l>=r) return;
    int m = l + (r-l)/2;
    mergeSort(arr, l, m);
    mergeSort(arr, m+1, r);
    merge(arr, l, m, r);
}

10. Palindrome Check

O(n)

Checks if string is same reversed.

bool isPal(string s) {
    string r = s;
    reverse(r.begin(), r.end());
    return s == r;
}

11. Heap Sort

O(n log n)

Sorting using Binary Heap.

void heapSort(int arr[], int n) {
    for(int i = n/2 - 1; i >= 0; i--) heapify(arr, n, i);
    for(int i = n - 1; i > 0; i--) {
        swap(arr[0], arr[i]);
        heapify(arr, i, 0);
    }
}

void DFS(int v) {
    visited[v] = true;
    cout << v << " ";
    for(int i : adj[v])
        if(!visited[i]) DFS(i);
}
